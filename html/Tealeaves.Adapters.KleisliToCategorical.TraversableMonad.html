          dist T G ∘ join (T := T) (A := G A) =
            map (F := G) join ∘ dist T G ∘ map (F := T) (dist T G).
    Proof.
      intros. unfold_ops @Dist_Bindt @Join_Bindt.
      do 2 rewrite <- bind_to_bindt.

      rewrite bindt_bind.
      unfold compose at 5.
      rewrite bind_bindt.
      rewrite fun_map_map.
      rewrite bindt_map.
      rewrite bind_ret.
      rewrite fun_map_id.
      reflexivity.
    Qed.

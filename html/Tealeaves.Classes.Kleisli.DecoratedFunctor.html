Class DecoratedFunctor (E: Type) (T: Type → Type) `{Mapd E T} :=
  { kdf_mapd1: ∀ (A: Type),
      mapd (extract (A := A)) = @id (T A);
    kdf_mapd2: ∀ (A B C: Type) (g: E × B → C) (f: E × A → B),
      mapd g ∘ mapd f = mapd (g ⋆1 f);
  }.

Section derived_instances.

  Context
    `{DecoratedFunctor E T}
    `{Map_T: Map T}
    `{! Compat_Map_Mapd E T}.

  Lemma map_mapd:
    ∀ (A B C: Type)
      (g: B → C)
      (f: E × A → B),
      map g ∘ mapd f = mapd (g ∘ f).
  Proof.
    intros.
    rewrite map_to_mapd.
    rewrite kdf_mapd2.
    rewrite kc1_01.
    reflexivity.
  Qed.

  Lemma mapd_map:
    ∀ (A B C: Type) (g: E × B → C) (f: A → B),
      mapd g ∘ map f = mapd (g ∘ map f).
  Proof.
    intros.
    rewrite map_to_mapd.
    rewrite kdf_mapd2.
    rewrite kc1_10.
    reflexivity.
  Qed.

  Lemma map_map: ∀ (A B C: Type) (f: A → B) (g: B → C),
      map g ∘ map f = map (F := T) (g ∘ f).
  Proof.
    intros.
    rewrite map_to_mapd.
    rewrite map_to_mapd.
    rewrite kdf_mapd2.
    rewrite kc1_00.
    rewrite map_to_mapd.
    reflexivity.
  Qed.

  Lemma map_id: ∀ (A: Type),
      map (@id A) = @id (T A).
  Proof.
    intros.
    rewrite map_to_mapd.
    change (id ∘ ?f) with f.
    rewrite kdf_mapd1.
    reflexivity.
  Qed.

End derived_instances.

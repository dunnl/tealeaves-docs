Definition dist_pair
  {b1 v1: Type}
  `{Map G} `{Mult G} `{Pure G}:
  list (G b1) × G v1 → G (list b1 × v1) :=
  fun '(x, y) ⇒ pure (@pair (list b1) v1) <⋆> dist list G x <⋆> y.

Definition dec_bag: ∀ (A: Type), list A → list (list A × A) :=
  fun A l ⇒ map (pair l) l.

Class ContainerMonad
  (T : Type → Type)
  `{Bind T T} `{Return T} `{ToSubset T} :=
  { contm_monad :> Monad T;
    contm_hom :> MonadHom T subset (@tosubset T _);
    contm_pointwise : ∀ (A B : Type) (t : T A) (f g : A → T B),
      (∀ a, a ∈ t → f a = g a) → bind f t = bind g t;
  }.

Class ContainerRightModule

      (∀ a, a ∈ t → f a = g a) → bind f t = bind g t;
  }.

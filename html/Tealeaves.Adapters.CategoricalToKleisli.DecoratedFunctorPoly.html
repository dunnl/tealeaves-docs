Lemma mapdp_to_categorical {T}
    `{Map_T: Map2 T}
    `{Decorate_T: DecoratePoly T}
    `{Mapdp_T: MapdPoly T}
    `{Compat: Compat_Mapdp_Categorical T}:
  ∀ (B1 B2 V1 V2: Type)
      (ρ: list B1 × B1 → B2)
      (σ: list B1 × V1 → V2)
      (t: T B1 V1),
      mapdp ρ σ t = map2 ρ σ (decp t).
Proof.
  now rewrite compat_mapdp_categorical.
Qed.

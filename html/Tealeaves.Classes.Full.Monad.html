Derived Categorical operations

Class Compat_Join_Bind
  (T : Type → Type)
  `{return_T: Return T}
  `{map_T: Map T}
  `{bind_T: Bind T T}
  `{join_T: Join T}
  : Prop :=
  { compat_join_bind: ∀ (A: Type),
      @join T join_T A = @bind T T bind_T (T A) A (@id (T A));
    compat_bind_join: ∀ (A B: Type) (f: A → T B),
      @bind T T bind_T A B f = @join T join_T B ∘ @map T map_T A (T B) f;
  }.

Lemma join_to_bind
  `{join_T: Join T}
  `{return_T: Return T}
  `{map_T: Map T}
  `{bind_T: Bind T T}
  `{compat: !Compat_Join_Bind T}:
  ∀ (A: Type),
    @join T join_T A =
      @bind T T bind_T (T A) A id.
Proof.
  intros. rewrite compat_join_bind.
  reflexivity.
Qed.

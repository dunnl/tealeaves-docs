Class Extract (W: Type -> Type) :=
  extract: W ‚áí (fun A => A).

  cojoin: W ‚áí W ‚àò W.

(** ** Typeclass *)

Class Comonad (W: Type -> Type)
  `{Map_W: Map W}
  `{Cojoin_W: Cojoin W}
  `{Extract_W: Extract W} :=
  { com_functor :> Functor W;
    com_extract_natural :> Natural (@extract W _);
    com_cojoin_natural :> Natural (@cojoin W _);
    com_extract_cojoin:
    `(extract (A := W A) ‚àò cojoin (A := A) = @id (W A));
    com_map_extr_cojoin:
    `(map (F := W) (extract (A := A)) ‚àò cojoin (A := A) = @id (W A));
    com_cojoin_cojoin:
    `(cojoin (A := W A) ‚àò cojoin (A := A) =
        map (F := W) (cojoin (A := A)) ‚àò cojoin (A := A));
  }.

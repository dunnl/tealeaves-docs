(** * Decorated Monads (Poly) *)
(**********************************************************************)
Class DecoratedMonadPoly
  (T: Type -> Type -> Type)
  `{Map2 T} `{DecoratePoly T}
  `{forall B, Return (T B)}
  `{forall B, Join (T B)} :=
  {
    dmp_functor :> Functor2 T;
    dmp_decorated :> DecoratedFunctorPoly T;
    dmp_monad :> forall B, Monad (T B);
    dmp_map_ret: forall B B' V V' (g: B -> B') (f: V -> V'),
      map2 g f ‚àò ret (T := T B) (A := V) =
        ret (T := T B') (A := V') ‚àò f;
    dmp_map_join: forall B B' V V' (g: B -> B') (f: V -> V'),
      map2 g f ‚àò join (T := T B) (A := V) =
        join (T := T B') (A := V') ‚àò map2 g (map2 g f);
    dmp_dec_ret: forall B V,
      decp ‚àò ret (T := T B) (A := V) =
        ret (T := T (Z B)) (A := Z2 B V) ‚àò ret (T := prod (list B));
    dmp_dec_join:
    forall (B V: Type),
      decp ‚àò join (T := T B) (A := V) =
        join (T := T (Z B)) ‚àò map2 id (shift2 ‚àò map_snd decp)
          ‚àò decp (B := B) (V := T B V);
  }.

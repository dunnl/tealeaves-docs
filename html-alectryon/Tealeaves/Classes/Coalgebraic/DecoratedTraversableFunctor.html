(** ** Operation <<cojoin_Batch3>> *)

Fixpoint cojoin_Batch3 `{ToBatch3 E T} {A B B' C: Type}
  (b: Batch (E * A) B C): Batch (E * A) B' (Batch (E * B') B C) :=
  match b with
  | Done c => Done (Done c)
  | Step rest (*:Batch (E * A) B (B -> C)*) (e, a)(*:E*A*) =>
      Step (map (F := Batch (E * A) B')

                 Step continue (e, b')
              ) (cojoin_Batch3 rest:
                Batch (E * A) B' (Batch (E * B') B (B -> C))))
        ((e, a): E * A)
  end.

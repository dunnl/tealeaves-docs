    #[export] Instance Dist_Mapdt: ApplicativeDist T :=
      fun G _ _ _ A =>
        mapdt (T := T) (G := G) (extract (W := (E √ó)) (A := G A)).

    #[export] Instance Decorate_Mapdt: Decorate E T :=
      fun A => mapdt (G := fun A => A) (@id (E * A)).

  End operations.
End DerivedOperations.
